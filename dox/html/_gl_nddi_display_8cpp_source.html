00108                                                                                         std::vector<int> start,
00109                                                                                         std::vector<int> end) {
00110         std::vector<int> position = start;
00111         bool fillFinished = false;
00112         
00113         // Move from start to end, filling in each location with the provided pixel
00114         do {
00115                 // Set coefficient matrix in coefficient plane at position

00137 void GlNddiDisplay::UpdateCoefficientMatrix(std::vector< std::vector<int> > coefficientMatrix, std::vector<int> location) {
00138         
00139         // Create a coefficient matrix array and initially fill it with with the
00140         // the current matrix at that location in the coefficient plane.
00141         int* matrix;
00142         matrix = (int*)malloc(CM_STORAGE);
00143         memcpy(matrix, &coefficientPlane_[(location[1] * displayWidth_ + location[0]) * CM_SIZE], CM_STORAGE);
00144         
00145         // Examine each coefficient in the coefficient matrix and use it unless it's a COFFICIENT_UNCHANGED
00146         for (int j = 0; j < CM_HEIGHT; j++) {
00147                 for (int i = 0; i < CM_WIDTH; i++) {
00148                         if (coefficientMatrix[i][j] != COFFICIENT_UNCHANGED) {
00149                                 matrix[j * CM_WIDTH + i] = coefficientMatrix[i][j];
00150                         }
00151                 }
00152         }
00153         
00154         // And copy the matrix back to the coefficient plane
00155         memcpy(&coefficientPlane_[(location[1] * displayWidth_ + location[0]) * CM_SIZE], matrix, CM_STORAGE);
00156         
00157         free(matrix);

00373 nddi::Pixel GlNddiDisplay::ComputePixel(int* iv) {
00374         
00375         // Grab the coefficient matrix and put it into an array for better SIMD later.
00376         int* coefficientMatrix = &coefficientPlane_[(iv[1] * displayWidth_ + iv[0]) * CM_SIZE];
00377 
00378         // Compute the position vector for the proper pixel in the frame volume.
00379         std::vector<int> fvPosition;
00380         // Matrix multiply the input vector by the coefficient matrix
00381         for (int j = 0; j < CM_HEIGHT; j++) {
00382                 fvPosition.push_back(0);
00383                 for (int i = 0; i < CM_WIDTH; i++) {
00384                         fvPosition[j] += coefficientMatrix[j * CM_WIDTH + i] * iv[i];
00385                 }
00386         }
00387         
00388         // Calculate the index into the frameVolume_ int array
00389         int fvArrayIndex = fvPosition[0];
00390         int dimensionalMultiplier = 1;
00391         for (int i = 1; i < frameVolumeDimensionalSizes_.size(); i++) {
00392                 dimensionalMultiplier *= frameVolumeDimensionalSizes_[i-1];
00393                 fvArrayIndex += fvPosition[i] * dimensionalMultiplier;
00394         }
00395         
00396         return frameVolume_[fvArrayIndex];
