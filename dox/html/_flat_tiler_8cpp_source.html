00055 double FlatTiler::UpdateDisplay(uint8_t* buffer, size_t width, size_t height)
00056 {
00057         double update_cost = 0.0f;
00058         int updates = 0;
00059         unsigned char mask = 0xff << (8 - bits_);
00060         
00061         // Break up the passed in buffer into one tile at a time
00062         for (int j_tile_map = 0; j_tile_map < tile_map_height_; j_tile_map++) {
00063                 for (int i_tile_map = 0; i_tile_map < tile_map_width_; i_tile_map++) {
00064                         // Initialize a tile's checksum
00065                         unsigned long tile_checksum = 0L;
00066                         
00067                         // Build the tile's pixel array while computing the checksum in an alternative tile which only
00068                         // holds the significant bits
00069                         nddi::Pixel tile_pixels[tile_width_ * tile_height_];
00070                         nddi::Pixel tile_pixels_sig_bits[tile_width_ * tile_height_];
00071                         
00072 #ifndef NO_OMP
00073 #pragma omp parallel for
00074 #endif // !NO_OMP
00075                         for (int j_tile = 0; j_tile < tile_height_; j_tile++) {
00076                                 
00077                                 // Compute the offset into the RGB buffer for this row in this tile
00078                                 int bufferOffset = 3 * ((j_tile_map * tile_height_ + j_tile) * width + (i_tile_map * tile_width_));
00079                                 
00080                                 for (int i_tile = 0; i_tile < tile_width_; i_tile++) {
00081                                         nddi::Pixel p, psb;
00082                                         
00083                                         // Just use a black pixel if our tile is hanging off the edge of the buffer
00084                                         if ((j_tile_map * tile_height_ + j_tile >= height) ||
00085                                                 (i_tile_map * tile_width_ + i_tile >= width) ) {
00086                                                 p.r = p.g = p.b = 0; p.a = 255;
00087                                         } else {
00088                                                 p.r = buffer[bufferOffset++];
00089                                                 p.g = buffer[bufferOffset++];
00090                                                 p.b = buffer[bufferOffset++];
00091                                                 p.a = 0xff;
00092                                         }
00093                                         tile_pixels[j_tile * tile_width_ + i_tile].packed = p.packed;
00094 
00095                                         psb.r = p.r & mask;
00096                                         psb.g = p.g & mask;
00097                                         psb.b = p.b & mask;
00098                                         psb.a = p.a & mask;
00099                                         tile_pixels_sig_bits[j_tile * tile_width_ + i_tile].packed = psb.packed;
00100                                 }
00101                         }
00102                         unsigned long crc = crc32(0L, Z_NULL, 0);
00103                         tile_checksum = crc32(crc, (unsigned char*)tile_pixels_sig_bits, tile_width_ * tile_height_ * sizeof(nddi::Pixel));
00104                         //tile_checksum = adler32(crc, (unsigned char*)tile_pixels_sig_bits, tile_width_ * tile_height_ * sizeof(nddi::Pixel));
00105                         
00106                         // If the checksum in the tile map doesn't match, then update the frame volume
00107                         if (tile_map_[i_tile_map][j_tile_map] != tile_checksum) {
00108                                 tile_map_[i_tile_map][j_tile_map] = tile_checksum;
00109                                 update_cost += UpdateFrameVolume(tile_pixels, i_tile_map, j_tile_map);
00110                                 updates++;
00111                         }
00112                 }
00113         }
00114 
00115         // Report update statistics
00116         tile_updates_ +=  updates;
00117         std::cout << "Tiles updated: " << updates << " Total updated: " << tile_updates_ << std::endl;
00118         
00119         return update_cost;
00120 }

00130 double FlatTiler::UpdateFrameVolume(nddi::Pixel* pixels, int i_map, int j_map) {
00131         
00132         double update_cost = 0.0f;
00133         
00134         // Setup start and end points
00135         std::vector<int> start, end;
00136         start.push_back(i_map * tile_width_); start.push_back(j_map * tile_height_);
00137         end.push_back(i_map * tile_width_ + tile_width_ - 1); end.push_back(j_map * tile_height_ + tile_height_ - 1);
00138         if (end[0] >= display_->DisplayWidth()) { end[0] = display_->DisplayWidth() - 1; }
00139         if (end[1] >= display_->DisplayHeight()) { end[1] = display_->DisplayHeight() - 1; }
00140         
00141         for (int j = 0; j < tile_height_; j++) {
00142                 start[1] = end[1] = j_map * tile_height_ + j;
00143                 if (end[1] >= display_->DisplayHeight()) { break; }
00144                 update_cost += display_->CopyPixels(&pixels[j * tile_width_], start, end);
00145         }
00146         return update_cost;
00147 }
